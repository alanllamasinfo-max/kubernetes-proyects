#####################################################################
#Si queremos hacerle una modificicaion al index deberomos aplicar estos comandos para ver reflejado el cambio

#Verifica el contenido del ConfigMap para ver si Kubernetes realmente tiene tu código HTML guardado:
kubectl get configmap web-content -o yaml

#Borra el ConfigMap viejo:
kubectl delete configmap web-content

#Crea el nuevo con el archivo corregido:
kubectl create configmap web-content --from-file=index.html

#Reinicia los Pods (porque Nginx no detecta cambios en archivos montados instantáneamente):
kubectl rollout restart deployment nginx-frontend
